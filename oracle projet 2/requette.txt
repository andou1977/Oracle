CREATE OR REPLACE FUNCTION rechsal
(
pempid number
)
return float
is rechsal float;
begin 
select salaire into rechsal from emp where empid =pempid;
return rechsal;
end;
/

DECLARE
  CURSOR mesemp IS
   SELECT * FROM Emp
   WHERE Dept = 'ADM';
  mes_emp emp%ROWTYPE;
BEGIN
  OPEN mesemp;
  LOOP
    FETCH mesemp INTO mes_emp;
    DBMS_OUTPUT.PUT_LINE(mes_emp.salaire);
    EXIT WHEN mesemp%NOTFOUND;
   END LOOP;
  CLOSE mesemp;
END;
